Programación del ESP32 con WiFi

En esta sesión se han probado los distintos protocolos para la comunicación por WiFi con el ESP32.

Anteriormente se ha instalado la placa ESP32 en Arduino IDE

# 1 Instalación

En primer lugar se instalaron las librerías necesarias:

  • https://github.com/me-no-dev/ESPAsyncWebServer.git
  • https://github.com/me-no-dev/AsyncTCP.git
  • https://github.com/bblanchon/ArduinoJson

# 2 Código de prueba

Codigo programado en arduino para ESP32

#include <ArduinoJson.h>
#include <WiFi.h>
#include <AsyncTCP.h>
#include <ESPAsyncWebServer.h>

//*** inicio definicion de variables ***//

const char *ssid = "nombre";            //aqui va el nombre del wifi
const char *password = "contraseña";    // aqui va la contraseña

//*** fin definicion de variables ***//

AsyncWebServer server(80);
AsyncWebSocket ws("/ws");

void handleWebSocketMessage(void *arg, uint8_t *data, size_t len)
{
    AwsFrameInfo *info = (AwsFrameInfo *)arg;
    if (info->final && info->index == 0 && info->len == len && info->opcode == WS_TEXT)
    {
        data[len] = 0;
        String str_test((char *)data);

        Serial.println(str_test);
        StaticJsonDocument<200> doc;
        DeserializationError error = deserializeJson(doc, str_test);

        // Test if parsing succeeds.
        if (error)
        {
            Serial.print(F("deserializeJson() failed: "));
            Serial.println(error.f_str());
            return;
        }

        //*** inicio código usuario ***//


        // cada vez que se pulsan las teclas (WASD) en la pagina
        // se ejecuta este código

        const char *sensor = doc["action"];
        Serial.println(sensor);

        //*** fin código usuario ***//
    }
}

void onEvent(AsyncWebSocket *server, AsyncWebSocketClient *client, AwsEventType type,
             void *arg, uint8_t *data, size_t len)
{
    switch (type)
    {
    case WS_EVT_CONNECT:
        Serial.printf("WebSocket client #%u connected from %s\n", client->id(), client->remoteIP().toString().c_str());
        break;
    case WS_EVT_DISCONNECT:
        Serial.printf("WebSocket client #%u disconnected\n", client->id());
        break;
    case WS_EVT_DATA:
        handleWebSocketMessage(arg, data, len);
        break;
    case WS_EVT_PONG:
    case WS_EVT_ERROR:
        break;
    }
}

void initWebSocket()
{
    ws.onEvent(onEvent);
    server.addHandler(&ws);
}

void setup()
{
    // Serial port for debugging purposes
    Serial.begin(115200);

    // Connect to Wi-Fi
    WiFi.begin(ssid, password);
    while (WiFi.status() != WL_CONNECTED)
    {
        delay(1000);
        Serial.println("Connecting to WiFi..");
    }

    // Print ESP Local IP Address
    Serial.println(WiFi.localIP());

    initWebSocket();

    // Route for root / web page
    server.on("/", HTTP_GET, [](AsyncWebServerRequest *request)
              { request->send_P(200, "text/html", "hi this works!"); });

    //*** inicio código usuario ***//

    //aqui se definen los tipos de los pines (input o output)

    //*** fin código usuario ***//

    server.begin();
}

long previousMillis = 0; // will store last time LED was updated
long interval = 1000;    // interval at which to blink (milliseconds)

void loop()
{
    ws.cleanupClients();

    unsigned long currentMillis = millis();

    //temporización de 1000ms
    if (currentMillis - previousMillis > interval)
    {

        //*** inicio código usuario ***//

        StaticJsonDocument<200> doc;

        doc["text"] = "hello";
        doc["timestamp"] = (int)(currentMillis/10000);

        char json_string[256];
        serializeJson(doc, json_string);
        ws.textAll(json_string);

        //*** fin código usuario ***//

        previousMillis = currentMillis;
    }
}
